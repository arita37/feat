300 string Feat::get_eqns(bool front)
301 {
302  vector<string> fields = {"id", "size", "complexity", "fitness",
303  "fitness_v", "coefficients", "pareto_rank", "eqn"};
304 
305  for (const auto& o : params.objectives)
306  {
307  if (!in(fields,o))
308  fields.push_back(o);
309  }
310 
311  vector<Individual>* printed_pop = NULL;
312 
313  string r = "";
314  for (int f = 0; f < fields.size(); ++f)
315  {
316  r += fields.at(f);
317  if (f < fields.size() -1)
318  r +="\t";
319  }
320  r += "\n";
321 
322  vector<size_t> idx;
323  bool subset = false;
324  if (front) // only return individuals on the Pareto front
325  {
326  if (use_arch)
327  {
328  printed_pop = &arch.archive;
329  }
330  else
331  {
332  unsigned n = 1;
333  subset = true;
334  idx = p_pop->sorted_front(n);
335  printed_pop = &p_pop->individuals;
336  }
337  }
338  else
339  printed_pop = &p_pop->individuals;
340 
341  if (!subset)
342  {
343  idx.resize(printed_pop->size());
344  std::iota(idx.begin(), idx.end(), 0);
345  }
346 
347  for (int i = 0; i < idx.size(); ++i)
348  {
349  Individual& ind = printed_pop->at(idx[i]);
350 
351  for (int f = 0; f < fields.size(); ++f)
352  {
353  if (fields.at(f) == "id")
354  r += to_string(ind.id);
355  else if (fields.at(f) == "size")
356  r += to_string(ind.size());
357  else if (fields.at(f) == "complexity")
358  r += to_string(ind.complexity());
359  else if (fields.at(f) == "fitness")
360  r += to_string(ind.fitness);
361  else if (fields.at(f) == "fitness_v")
362  r += to_string(ind.fitness_v);
363  else if (fields.at(f) == "coefficients")
364  {
365  auto tmpw = ind.ml->get_weights();
366  for (int w = 0; w < tmpw.size(); ++w)
367  {
368  r += to_string(tmpw.at(w));
369  if (w != tmpw.size()-1)
370  r += ",";
371  }
372  }
373  else if (fields.at(f) == "pareto_rank")
374  r += to_string(ind.rank);
375  else if (fields.at(f) == "eqn")
376  r += ind.get_eqn();
377 
378  if (f < fields.size() - 1)
379  r += "\t";
380  }
381  if (i < idx.size() -1)
382  r += "\n";
383  }
384  printed_pop = NULL;
385  delete printed_pop;
386 
387  return r;
388 }

660  arch.archive = p_pop->individuals;
661  }
662 
663  logger.log("\n===\nRun Completed. Total time taken is "

665

667  log.close();

670 void Feat::run_generation(unsigned int g,
671  vector<size_t> survivors,
672  DataRef &d,
673  std::ofstream &log,
674  float fraction,
675  unsigned& stall_count)
676 {
677  d.t->set_protected_groups();
678 
679  params.set_current_gen(g);
680 
681  // select parents
682  logger.log("selection..", 2);
683  vector<size_t> parents = p_sel->select(*p_pop, F, params, *d.t);
684  logger.log("parents:\n"+p_pop->print_eqns(), 3);
685 
686  // variation to produce offspring
687  logger.log("variation...", 2);
688  p_variation->vary(*p_pop, parents, params,*d.t);
689  logger.log("offspring:\n" + p_pop->print_eqns(true), 3);
690 
691  // evaluate offspring
692  logger.log("evaluating offspring...", 2);
693  p_eval->fitness(p_pop->individuals, *d.t, F, params, true);
694  p_eval->validation(p_pop->individuals, *d.v, params, true);
695 
696  // select survivors from combined pool of parents and offspring
697  logger.log("survival...", 2);
698  survivors = p_surv->survive(*p_pop, F, params, *d.t);
699 
700  // reduce population to survivors
701  logger.log("shrinking pop to survivors...",2);
702  p_pop->update(survivors);
703  logger.log("survivors:\n" + p_pop->print_eqns(), 3);
704 
705  logger.log("update best...",2);
706  bool updated_best = update_best(d);
707 
708  logger.log("calculate stats...",2);
709  calculate_stats(d);
710 
711  if (params.max_stall > 0)
712  update_stall_count(stall_count, updated_best);
713 
714  logger.log("update archive...",2);
715  if (use_arch)
716  arch.update(*p_pop,params);
717 
718  if(params.verbosity>1)
719  print_stats(log, fraction);
720  else if(params.verbosity == 1)
721  printProgress(fraction);
722 
723  if (print_pop > 1 || print_pop > 0 && g == params.gens-1)
724  print_population();
725 
726  // tighten learning rate for grad descent as evolution progresses
727  if (params.backprop)
728  {
729  params.bp.learning_rate = \
730  (1-1/(1+float(params.gens)))*params.bp.learning_rate;
731  logger.log("learning rate: "
732  + std::to_string(params.bp.learning_rate),3);
733  }
734  logger.log("finished with generation...",2);
735 
736 }

793 void Feat::simplify_model(DataRef& d, Individual& ind)
794 {
795  /* Simplifies the final model using some expert rules and stochastic hill
796  * climbing.
797  * Expert rules:
798  * - NOT(NOT(x)) simplifies to x
799  * Stochastic hill climbing:
800  * for some number iterations, apply delete mutation to the equation.
801  * if the output of the model doesn't change, keep the mutations.
802  */
803 
805  // check for specific patterns
807  //
808  Individual tmp_ind = ind;
809  int starting_size = ind.size();
810  vector<size_t> roots = tmp_ind.program.roots();
811  vector<size_t> idx_to_remove;
812 
813  logger.log("\n=========\ndoing pattern pruning...",2);
814 
815  for (auto r : roots)
816  {
817  /* cout << "r: " << r << "\n"; */
818  size_t start = tmp_ind.program.subtree(r);
819  int first_occurence = -2;
820 
821  /* cout << "start: " << start << "\n"; */
822  for (int i = start ; i <= r; ++i)
823  {
824  /* cout << "i: " << i << ", first_occurence: " << first_occurence */
825  /* << "\n"; */
826  if (tmp_ind.program.at(i)->name.compare("not")==0)
827  {
828  if (first_occurence == i-1) // indicates two NOTs in a row
829  {
830  /* cout << "pushing back " << first_occurence */
831  /* << " and " << i << " to idx_to_remove\n"; */
832  idx_to_remove.push_back(first_occurence);
833  idx_to_remove.push_back(i);
834  // reset first_occurence so we don't pick up triple nots
835  first_occurence = -2;
836  }
837  else
838  {
839  first_occurence = i;
840  }
841  }
842  }
843  }
844  // remove indices in reverse order so they don't change
845  std::reverse(idx_to_remove.begin(), idx_to_remove.end());
846  for (auto idx: idx_to_remove)
847  {
848  /* cout << "removing " << tmp_ind.program.at(idx)->name */
849  /* << " at " << idx << "\n"; */
850  tmp_ind.program.erase(tmp_ind.program.begin()+idx);
851  }
852  int end_size = tmp_ind.size();
853  logger.log("pattern pruning reduced best model size by "
854  + to_string(starting_size - end_size)
855  + " nodes\n=========\n", 1);
856  if (tmp_ind.size() < ind.size())
857  ind = tmp_ind;
858 
860  // prune dimensions
862  /* set_verbosity(3); */
863  int iterations = ind.get_dim();
864  logger.log("\n=========\ndoing correlation deletion mutations...",2);
865  starting_size = ind.size();
866  float tolerance = 0.001;
867  for (int i = 0; i < iterations; ++i)
868  {
869  Individual tmp_ind = ind;
870  bool perfect_correlation = p_variation->correlation_delete_mutate(
871  tmp_ind, ind.Phi, params, *d.o);
872 
873  if (ind.size() == tmp_ind.size())
874  {
875  continue;
876  }
877  bool pass = true;
878  shared_ptr<CLabels> yhat = tmp_ind.fit(*d.o, params, pass);
879 
880 
881  if (((ind.yhat - tmp_ind.yhat).norm()/ind.yhat.norm()
882  <= tolerance )
883  or perfect_correlation)
884  {
885  logger.log("\ndelete dimension mutation success: went from "
886  + to_string(ind.size()) + " to "
887  + to_string(tmp_ind.size()), 3);
888  ind = tmp_ind;
889  }
890  else
891  {
