124  void set_term_weights(const vector<double>& w)
125  {
126  //assert(w.size()==terminals.size());
127  string weights;
128  double u = 1.0/double(w.size());
129  term_weights.clear();
130  vector<double> sw = softmax(w);
131  int x = 0;
132  for (unsigned i = 0; i < terminals.size(); ++i)
133  {
134  if(terminals[i]->otype == 'z')
135  term_weights.push_back(u);
136  else
137  {
138  term_weights.push_back(u + feedback*(sw[x]-u));
139  x++;
140  }
141  }
142 
143  weights = "term weights: ";
144  for (auto tw : term_weights)
145  weights += std::to_string(tw)+" ";
146  weights += "\n";
147 
148  msg(weights, 2);
