97  void set_term_weights(const vector<double>& w)
98  {
99  assert(w.size()==terminals.size());
100  string weights;
101  double u = 1.0/double(w.size());
102  term_weights.clear();
103  vector<double> sw = softmax(w);
104  for (unsigned i = 0; i<sw.size(); ++i)
105  term_weights.push_back(u + feedback*(sw[i]-u));
106 
107  weights = "term weights: ";
108  for (auto tw : term_weights)
109  weights += std::to_string(tw)+" ";
110  weights += "\n";
111 
112  msg(weights, 2);
113  }
