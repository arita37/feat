14  VectorXd log_loss(const VectorXd& y, const VectorXd& yhat)
15  {
16  /* std::cout << "in log loss\n"; */
17  double eps = pow(10,-10);
18 
19  VectorXd loss(y.rows());
20  for (unsigned i = 0; i < y.rows(); ++i)
21  {
22  if (yhat(i) < eps || 1 - yhat(i) < eps)
23  // clip probabilities since log loss is undefined for yhat=0 or yhat=1
24  loss(i) = -(y(i)*log(eps) + (1-y(i))*log(1-eps));
25  else
26  loss(i) = -(y(i)*log(yhat(i)) + (1-y(i))*log(1-yhat(i)));
27  }
28  return loss;
29  }

40  double bal_zero_one_loss(const VectorXd& y, const VectorXd& yhat, VectorXd& loss )
41  {

43  vector<double> uc = unique(y);

45  for (const auto& i : uc)

48  // sensitivity (TP) and specificity (TN)
49  vector<double> TP(c.size(),0.0), TN(c.size(), 0.0), P(c.size(),0.0), N(c.size(),0.0);
50  ArrayXd class_accuracies(c.size());
51 
52  // get class counts
53 
54  for (unsigned i=0; i< c.size(); ++i)
55  {
56  P.at(i) = (y.array().cast<int>() == c[i]).count(); // total positives for this class
57  N.at(i) = (y.array().cast<int>() != c[i]).count(); // total negatives for this class
58  }
59 
60 
61  for (unsigned i = 0; i < y.rows(); ++i)
62  {
63  if (yhat(i) == y(i)) // true positive
64  ++TP.at(y(i) == -1 ? 0 : y(i)); // if-then ? accounts for -1 class encoding
65 
66  for (unsigned j = 0; j < c.size(); ++j)
67  if ( y(i) !=c.at(j) && yhat(i) != c.at(j) ) // true negative
68  ++TN.at(j);
69 
70  }
71 
72  // class-wise accuracy = 1/2 ( true positive rate + true negative rate)
73  for (unsigned i=0; i< c.size(); ++i){
74  class_accuracies(i) = (TP[i]/P[i] + TN[i]/N[i])/2;
75  //std::cout << "TP(" << i << "): " << TP[i] << ", P[" << i << "]: " << P[i] << "\n";

78  }
