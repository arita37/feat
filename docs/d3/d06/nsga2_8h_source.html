60  vector<size_t> NSGA2::survive(Population& pop, const MatrixXd& F, const Parameters& params)
61  {
62  /* Selection using the survival scheme of NSGA-II.
63  *
64  * Input:
65  *
66  * pop: population of programs.
67  * params: parameters.
68  * r: random number generator
69  *
70  * Output:
71  *
72  * selected: vector of indices corresponding to columns of F that are selected.
73  * modifies individual ranks, objectives and dominations.
74  */
75 
76  // set objectives
77  #pragma omp parallel for
78  for (unsigned int i=0; i<pop.size(); ++i)
79  pop.individuals[i].set_obj(params.objectives);
80 
81  // fast non-dominated sort
82  fast_nds(pop);
83 
84  // Push back selected individuals until full
85  vector<size_t> selected;
86  int i = 0;
87  while ( selected.size() + front[i].size() < params.pop_size)
88  {
89  std::vector<int>& Fi = front[i]; // indices in front i
90  //crowding_distance(i); // calculate crowding in Fi
91 
92  for (int j = 0; j < Fi.size(); ++j) // Pt+1 = Pt+1 U Fi
93  selected.push_back(Fi[j]);
94 
95  ++i;
96  }
97 
98  crowding_distance(pop,i); // calculate crowding in final front to include
99  std::sort(front[i].begin(),front[i].end(),sort_n(pop));
100 
101  const int extra = params.pop_size - selected.size();
102  for (int j = 0; j < extra; ++j) // Pt+1 = Pt+1 U Fi[1:N-|Pt+1|]
103  selected.push_back(front[i][j]);
104 
105  return selected;
106  }

108  void NSGA2::fast_nds(Population& pop)
109  {
110  front.resize(1);
111  front[0].clear();
112  //std::vector< std::vector<int> > F(1);
113  #pragma omp parallel for
114  for (int i = 0; i < pop.size(); ++i) {
115 
116  std::vector<unsigned int> dom;
117  int dcount = 0;
118 
119  Individual& p = pop.individuals[i];
120  // p.dcounter = 0;
121  // p.dominated.clear();
122 
123  for (int j = 0; j < pop.size(); ++j) {
124 
125  Individual& q = pop.individuals[j];
126 
127  int compare = p.check_dominance(q);
128  if (compare == 1) { // p dominates q
129  //p.dominated.push_back(j);
130  dom.push_back(j);
131  } else if (compare == -1) { // q dominates p
132  //p.dcounter += 1;
133  dcount += 1;
134  }
135  }
136 
137  #pragma omp critical
138  {
139  p.dcounter = dcount;
140  p.dominated.clear();
141  p.dominated = dom;
142 
143 
144  if (p.dcounter == 0) {
145  p.set_rank(1);
146  front[0].push_back(i);
147  }
148  }
149 
150  }
151 
152  // using OpenMP can have different orders in the front[0]
153  // so let's sort it so that the algorithm is deterministic
154  // given a seed
155  std::sort(front[0].begin(), front[0].end());
156 
157  int fi = 1;
158  while (front[fi-1].size() > 0) {
159 
160  std::vector<int>& fronti = front[fi-1];
161  std::vector<int> Q;
162  for (int i = 0; i < fronti.size(); ++i) {
163 
164  Individual& p = pop.individuals[fronti[i]];
165 
166  for (int j = 0; j < p.dominated.size() ; ++j) {
167 
168  Individual& q = pop.individuals[p.dominated[j]];
169  q.dcounter -= 1;
170 
171  if (q.dcounter == 0) {
172  q.set_rank(fi+1);
173  Q.push_back(p.dominated[j]);
174  }
175  }
176  }
177 
178  fi += 1;
179  front.push_back(Q);
180  }

184  void NSGA2::crowding_distance(Population& pop, int fronti)
185  {
186  std::vector<int> F = front[fronti];
187  if (F.size() == 0 ) return;
188 
189  const int fsize = F.size();
190 
191  for (int i = 0; i < fsize; ++i)
192  pop.individuals[F[i]].crowd_dist = 0;
193 
194 
195  const int limit = pop[0].obj.size();
196  for (int m = 0; m < limit; ++m) {
197 
198  std::sort(F.begin(), F.end(), comparator_obj(pop,m));
199 
200  // in the paper dist=INF for the first and last, in the code
201  // this is only done to the first one or to the two first when size=2
202  pop.individuals[F[0]].crowd_dist = std::numeric_limits<double>::max();
203  if (fsize > 1)
204  pop.individuals[F[fsize-1]].crowd_dist = std::numeric_limits<double>::max();
205 
206  for (int i = 1; i < fsize-1; ++i)
207  {
208  if (pop.individuals[F[i]].crowd_dist != std::numeric_limits<double>::max())
209  { // crowd over obj
210  pop.individuals[F[i]].crowd_dist +=
211  (pop.individuals[F[i+1]].obj[m] - pop.individuals[F[i-1]].obj[m])
212  / (pop.individuals[F[fsize-1]].obj[m] - pop.individuals[F[0]].obj[m]);
213  }
214  }
215  }
216  }
