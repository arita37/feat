299  void Feat::fit(MatrixXd& X, VectorXd& y)
300  {
320  // start the clock
321  timer.Reset();
322 
323  if(str_dim.compare("") != 0)
324  {
325  string dimension;
326  dimension = str_dim.substr(0, str_dim.length() - 1);
327  params.msg("STR DIM IS "+ dimension, 1);
328  params.msg("Cols are " + std::to_string(X.rows()), 1);
329  params.msg("Setting dimensionality as " +
330  std::to_string((int)(ceil(stod(dimension)*X.rows()))), 1);
331  set_max_dim(ceil(stod(dimension)*X.rows()));
332  }
333 
334  params.check_ml();
335 
336  if (params.classification) // setup classification endpoint
337  params.set_classes(y);
338 
339  set_dtypes(find_dtypes(X));
340 
341  p_ml = make_shared<ML>(params); // intialize ML
342  p_pop = make_shared<Population>(params.pop_size);
343 
344  // split data into training and test sets
345  MatrixXd X_t(X.rows(),int(X.cols()*params.split));
346  MatrixXd X_v(X.rows(),int(X.cols()*(1-params.split)));
347  VectorXd y_t(int(y.size()*params.split)), y_v(int(y.size()*(1-params.split)));
348  train_test_split(X,y,X_t,X_v,y_t,y_v,params.shuffle);
349 
350  // define terminals based on size of X
351  params.set_terminals(X.rows());
352 
353  // initial model on raw input
354  params.msg("Fitting initial model", 1);
355  initial_model(X_t,y_t);
356  params.msg("Initial score: " + std::to_string(best_score), 1);
357 
358  // initialize population
359  params.msg("Initializing population", 1);
360  p_pop->init(best_ind,params);
361  params.msg("Initial population:\n"+p_pop->print_eqns(),2);
362 
363  // resize F to be twice the pop-size x number of samples
364  F.resize(X_t.cols(),int(2*params.pop_size));
365 
366  // evaluate initial population
367  params.msg("Evaluating initial population",1);
368  p_eval->fitness(*p_pop,X_t,y_t,F,params);
369 
370  vector<size_t> survivors;
371 
372  // main generational loop
373  for (unsigned int g = 0; g<params.gens; ++g)
374  {
375  // select parents
376  params.msg("selection..", 2);
377  vector<size_t> parents = p_sel->select(*p_pop, F, params);
378  params.msg("parents:\n"+p_pop->print_eqns(), 2);
379 
380  // variation to produce offspring
381  params.msg("variation...", 2);
382  p_variation->vary(*p_pop, parents, params);
383  params.msg("offspring:\n" + p_pop->print_eqns(true), 2);
384 
385  // evaluate offspring
386  params.msg("evaluating offspring...", 2);
387  p_eval->fitness(*p_pop, X_t, y_t, F, params, true);
388 
389  // select survivors from combined pool of parents and offspring
390  params.msg("survival...", 2);
391  survivors = p_surv->survive(*p_pop, F, params);
392 
393  // reduce population to survivors
394  params.msg("shrinking pop to survivors...",2);
395  p_pop->update(survivors);
396  params.msg("survivors:\n" + p_pop->print_eqns(), 2);
397 
398  update_best();
399  if (params.verbosity>0) print_stats(g+1);
400  }
401  params.msg("finished",1);
402  params.msg("best training representation: " + best_ind.get_eqn(),1);
403  params.msg("train score: " + std::to_string(best_score), 1);
404  // evaluate population on validation set
405  if (params.split < 1.0)
406  {
407  F_v.resize(X_v.cols(),int(2*params.pop_size));
408  p_eval->fitness(*p_pop, X_v, y_v, F_v, params);
409  initial_model(X_v, y_v); // calculate baseline model validation score
410  update_best(); // get the best validation model
411  }
412 
413  final_model(X,y); // fit final model to best model
414  params.msg("best validation representation: " + best_ind.get_eqn(),1);
415  params.msg("validation score: " + std::to_string(best_score), 1);
416 
417 
418  // write model to file
419  std::ofstream out_model;
420  out_model.open("model_" + name + ".txt");
421  out_model << best_ind.get_eqn() ;
422  out_model.close();
423  }

479  MatrixXd Phi = transform(X);

481  SGVector<double> y_pred;
482  VectorXd yhat;
483 
484  if (params.classification && params.n_classes == 2 &&
485  (!params.ml.compare("SVM") || !params.ml.compare("LR")))
486  {
487  auto tmp = p_ml->p_est->apply_binary(PhiSG);
488  y_pred = tmp->get_labels();
489  delete tmp;
490 
491  }
492  else if (params.classification)
493  {
494  auto tmp = p_ml->p_est->apply_multiclass(PhiSG);
495  y_pred = tmp->get_labels();
496  delete tmp;
497  }
498  else
499  {
500  auto tmp = p_ml->p_est->apply_regression(PhiSG);
501  y_pred = tmp->get_labels();
502  delete tmp;
503  }
504 
505  yhat = Eigen::Map<VectorXd>(y_pred.data(),y_pred.size());
506 
507  if (params.classification && (!params.ml.compare("LR") || !params.ml.compare("SVM")))
508  // convert -1 to 0
509  yhat = (yhat.cast<int>().array() == -1).select(0,yhat);
510 
511  return yhat;
