151  bool reverse)
152  {
153  if (c.empty()) // determine unique class values
154  {
155  vector<double> uc = unique(y);
156  for (const auto& i : uc)
157  c.push_back(int(i));
158  }
159 
160  // sensitivity (TP) and specificity (TN)
161  vector<double> TP(c.size(),0.0), TN(c.size(), 0.0), P(c.size(),0.0), N(c.size(),0.0);
162  ArrayXd class_accuracies(c.size());
163 
164  // get class counts
165 
166  for (unsigned i=0; i< c.size(); ++i)
167  {
168  P.at(i) = (y.array().cast<int>() == c[i]).count(); // total positives for this class
169  N.at(i) = (y.array().cast<int>() != c[i]).count(); // total negatives for this class
170  }
171 
172 
173  for (unsigned i = 0; i < y.rows(); ++i)
174  {
175  if (yhat(i) == y(i)) // true positive
176  ++TP.at(y(i) == -1 ? 0 : y(i)); // if-then ? accounts for -1 class encoding
177 
178  for (unsigned j = 0; j < c.size(); ++j)
179  if ( y(i) !=c.at(j) && yhat(i) != c.at(j) ) // true negative
180  ++TN.at(j);
181 
182  }
183 
184  // class-wise accuracy = 1/2 ( true positive rate + true negative rate)
185  for (unsigned i=0; i< c.size(); ++i){
186  class_accuracies(i) = (TP[i]/P[i] + TN[i]/N[i])/2;
187  //std::cout << "TP(" << i << "): " << TP[i] << ", P[" << i << "]: " << P[i] << "\n";
188  //std::cout << "TN(" << i << "): " << TN[i] << ", N[" << i << "]: " << N[i] << "\n";
189  //std::cout << "class accuracy(" << i << "): " << class_accuracies(i) << "\n";
190  }
191  if (reverse)
