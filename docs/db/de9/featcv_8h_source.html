171  void FeatCV::parse()
172  {
173  int startIndex = hyperParams.find('{');
174  int endIndex = hyperParams.find("}");
175 
176  string curParams;
177  string token, values;
178  tokenType type;
179 
180  int tokenStart, tokenEnd, valueStart, valueEnd;
181 
182  while(startIndex != string::npos && endIndex != string::npos)
183  {
184  clearVectors();
185 
186  curParams = trim(hyperParams.substr(startIndex+1, endIndex - startIndex - 1));
187 
188 
189  tokenStart = curParams.find('\'');
190  tokenEnd = curParams.find('\'', tokenStart + 1);
191  valueStart = curParams.find('(', tokenEnd);
192  valueEnd = curParams.find(')', valueStart);
193 
194  while(tokenStart != string::npos && tokenEnd != string::npos && valueStart != string::npos && valueEnd != string::npos)
195  {
196  token = trim(curParams.substr(tokenStart + 1, tokenEnd - tokenStart - 1));
197  values = trim(curParams.substr(valueStart + 1, valueEnd - valueStart - 1));
198 
199  type = getTokenType(token);
200 
201  switch(type)
202  {
203  case population : fillVector(pop_size, values, Integer); break;
204  case generation : fillVector(gens, values, Integer); break;
205  case ml : fillVector(mlStr, values); break;
206  case maxStall : fillVector(max_stall, values, Integer); break;
207  case selection : fillVector(sel, values); break;
208  case survival : fillVector(surv, values); break;
209  case crossRate : fillVector(cross_rate, values, Float); break;
210  case functions : fillVector(funcs, values); break;
211  case maxDepth : fillVector(max_depth, values, UInteger); break;
212  case maxDim : fillVector(max_dim, values, UInteger); break;
213  case erc : fillVector(ercVec, values, Bool); break;
214  case objectives : fillVector(obj, values); break;
215  case feedBack : fillVector(fb, values, Double); break;
216  case unknown: cout<<"Unknown token type "<<token<<std::endl; throw;
217  }
218 
219  tokenStart = curParams.find('\'', valueEnd + 1);
220  tokenEnd = curParams.find('\'', tokenStart + 1);
221  valueStart = curParams.find('(', tokenEnd + 1);
222  valueEnd = curParams.find(')', valueStart + 1);
223 
224  }
225 
226  setDefaults();
227 
228  /*for(auto str : mlStr)
229  cout<<"Ml method "<<str<<"\n";*/
230 
231  pushObjects();
232 
233  startIndex = hyperParams.find('{', endIndex + 1);
234  endIndex = hyperParams.find("}", endIndex + 1);
235 
236  }
237 
238  clearVectors();
239  }

406  void FeatCV::fit(MatrixXd x, VectorXd &y)
407  {
408  parse();
409  create_folds(x.cols());
410 
411  cout<<"Total number of objects created are "<<featObjs.size()<<"\n";
412 
413  #pragma omp parallel
414  {
415  #pragma omp single
416  for(int i = 0; i < featObjs.size(); i++)
417  {
418  #pragma omp task
419  {
420  VectorXd objScore(foldSize);
421 
422  int testIndex = foldSize - 1;
423 
424  // loop to fit all data folds in a single object and calculate the mean score
425  for(int j = 0; j < foldSize; j++)
426  {
427  int filled = 0, k, l;
428 
429  MatrixXd trainX(x.rows(), x.cols() - dataFolds[testIndex].quantity);
430  VectorXd trainY(x.cols() - dataFolds[testIndex].quantity);
431 
432  //creating training data from data folds
433  for(k = 0; k < foldSize; k++)
434  {
435  if(k != testIndex)
436  {
437  trainX.block(0, filled, x.rows(), dataFolds[k].quantity) =
438  x.block(0, dataFolds[k].startIndex, x.rows(), dataFolds[k].quantity);
439  filled += dataFolds[k].quantity;
440  }
441  }
442 
443  // creating training data outputs from data folds
444  for(k = 0, l = 0; k < y.size(); k++)
445  if(k < dataFolds[testIndex].startIndex ||
446  k >= dataFolds[testIndex].startIndex+dataFolds[testIndex].quantity)
447  trainY(l++) = y(k);
448 
449  // set dtypes for training data and train the model
450  featObjs[i].obj.set_dtypes(find_dtypes(trainX));
451  featObjs[i].obj.fit(trainX, trainY);
452 
453  // extracting testdata and actual values for test data from data folds
454  MatrixXd testData(x.rows(), dataFolds[testIndex].quantity);
455  VectorXd actualValues(dataFolds[testIndex].quantity);
456 
457  testData <<
458  x.block(0, dataFolds[testIndex].startIndex,x.rows(),dataFolds[testIndex].quantity);
459 
460  for(k = 0; k < dataFolds[testIndex].quantity; k++)
461  actualValues(k) = y(dataFolds[testIndex].startIndex+k);
462 
463  // prediction on test data
464  VectorXd prediction = featObjs[i].obj.predict(testData);
465 
466  // calculating difference between actual and predicted values
467  if (featObjs[i].obj.get_classification())
468  // use classification accuracy
469  objScore[j] = ((prediction.cast<int>().array() !=
470  actualValues.cast<int>().array()).cast<double>()).mean();
471  else
472  // use mean squared error
473  objScore[j] = ((prediction - actualValues).array().pow(2)).mean();
474 
475  testIndex = (testIndex+1)%foldSize;
476 
477  }
478 
479  // setting mean score for the model based on the tuning parameters
480  featObjs[i].score = objScore.mean();
481  cout<<"****Finished object "<<i<<"\n";
482  }
483 
484  }
485  }
486 
487  bestScoreIndex = 0;
488 
489  // finding the best model from validation and printing its parameters
490  for(int i = 1; i < featObjs.size(); i++)
491  if(featObjs[i].score < featObjs[bestScoreIndex].score)
492  bestScoreIndex = i;
493 
494  cout << "\n********************\nBest tuning parameters for this data is \n";
495  cout << "\nML = " << featObjs[bestScoreIndex].obj.get_ml();
496  cout << "\nPopulation size = " << featObjs[bestScoreIndex].obj.get_pop_size();
497  cout << "\nGenerations = " << featObjs[bestScoreIndex].obj.get_generations();
498  cout << "\nDimensions = " << featObjs[bestScoreIndex].obj.get_max_dim();
499  cout << "\nDepth = " << featObjs[bestScoreIndex].obj.get_max_depth();
500  cout << "\nCross Rate = " <<featObjs[bestScoreIndex].obj.get_cross_rate();
501  cout << "\nScore = " << featObjs[bestScoreIndex].score<<"\n";
502 
503  }
