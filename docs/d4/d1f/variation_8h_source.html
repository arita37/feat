361  bool Variation::cross(Individual& mom, Individual& dad, Individual& child,
362  const Parameters& params)
363  {
375  bool subtree = r() <0.5; // half the time, do subtree xo.
376  // half the time, swap dimensions.
377  vector<size_t> mlocs, dlocs; // mom and dad locations for consideration
378  size_t i1, j1, i2, j2; // i1-j1: mom portion, i2-j2: dad portion
379 
380  if (subtree)
381  {
382  params.msg("\tsubtree xo",2);
383  // limit xo choices to matching output types in the programs.
384  vector<char> d_otypes;
385  for (const auto& p : dad.program)
386  if(!in(d_otypes,p->otype))
387  d_otypes.push_back(p->otype);
388 
389  // get valid subtree locations
390  for (size_t i =0; i<mom.size(); ++i)
391  if (in(d_otypes,mom.program[i]->otype))
392  mlocs.push_back(i);
393  if (mlocs.size()==0) // mom and dad have no overlapping types, can't cross
394  {
395  params.msg("\tno overlapping types between " + mom.program_str() + ","
396  + dad.program_str() + "\n", 1);
397  return 0;
398  }
399  j1 = r.random_choice(mlocs,mom.get_p(mlocs));
400 
401  // get locations in dad's program that match the subtree type picked from mom
402  for (size_t i =0; i<dad.size(); ++i)
403  if (dad.program[i]->otype == mom.program[j1]->otype) dlocs.push_back(i);
404  }
405  else // half the time, pick a root node
406  {
407  params.msg("\troot xo",2);
408  mlocs = mom.program.roots();
409  dlocs = dad.program.roots();
410  params.msg("\t\trandom choice mlocs (size "+std::to_string(mlocs.size())+")",2);
411  j1 = r.random_choice(mlocs,mom.get_p(mlocs)); // weighted probability choice
412  }
413  // get subtree
414  i1 = mom.program.subtree(j1);
415 
416  // get dad subtree
417  j2 = r.random_choice(dlocs);
418  i2 = dad.program.subtree(j2);
419 
420  // make child program by splicing mom and dad
421  splice_programs(child.program, mom.program, i1, j1, dad.program, i2, j2 );
422 
423  if (params.verbosity >= 2)
424  print_cross(mom,i1,j1,dad,i2,j2,child);
425 
426  assert(is_valid_program(child.program,params.num_features, params.longitudinalMap));
427  // check child depth and dimensionality
428  return child.size()>0 && child.size() <= params.max_size
429  && child.get_dim() <= params.max_dim;
430  }
