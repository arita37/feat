143  void make_tree(NodeVector& program,
144  const NodeVector& functions,
145  const NodeVector& terminals, int max_d,
146  const vector<double>& term_weights, char otype)
147  {
148 
153  if (max_d == 0 || r.rnd_flt() < terminals.size()/(terminals.size()+functions.size()))
154  {
155  // append terminal
156  vector<size_t> ti; // indices of valid terminals
157  vector<double> tw; // weights of valid terminals
158  for (size_t i = 0; i<terminals.size(); ++i)
159  {
160  if (terminals[i]->otype == otype) // grab terminals matching output type
161  {
162  ti.push_back(i);
163  tw.push_back(term_weights[i]);
164  }
165  }
166  auto t = terminals[r.random_choice(ti,tw)]->clone();
167  //std::cout << t->name << " ";
168  program.push_back(t->clone());
169  }
170  else
171  {
172  // let fi be indices of functions whose output type matches otype and, if max_d==1,
173  // with no boolean inputs (assuming all input data is floating point)
174  vector<size_t> fi;
175  for (size_t i = 0; i<functions.size(); ++i)
176  if (functions[i]->otype==otype && (max_d>1 || functions[i]->arity['b']==0)
177  && (max_d>1 || functions[i]->arity['z']==0))
178  fi.push_back(i);
179 
180  if (fi.size()==0){
181  if(otype == 'z')
182  {
183  make_tree(program, functions, terminals, 0, term_weights, 'z');
184  return;
185  }
186  else{
187  std::cout << "---\n";
188  std::cout << "f1.size()=0. current program: ";
189  for (const auto& p : program) std::cout << p->name << " ";
190  std::cout << "\n";
191  std::cout << "otype: " << otype << "\n";
192  std::cout << "max_d: " << max_d << "\n";
193  std::cout << "functions: ";
194  for (const auto& f: functions) std::cout << f->name << " ";
195  std::cout << "\n";
196  std::cout << "---\n";
197  }
198  }
199 
200  assert(fi.size() > 0 && "The operator set specified results in incomplete programs.");
201 
202  // append a random choice from fs
203  auto t = functions[r.random_choice(fi)]->clone();
204  //std::cout << t->name << " ";
205  program.push_back(t->clone());
206 
207  std::unique_ptr<Node> chosen(program.back()->clone());
208  // recurse to fulfill the arity of the chosen function
209  for (size_t i = 0; i < chosen->arity['f']; ++i)
210  make_tree(program, functions, terminals, max_d-1, term_weights,'f');
211  for (size_t i = 0; i < chosen->arity['b']; ++i)
212  make_tree(program, functions, terminals, max_d-1, term_weights, 'b');
213  for (size_t i = 0; i < chosen->arity['z']; ++i)
214  make_tree(program, functions, terminals, max_d-1, term_weights, 'z');
215  }
216 
217  }
