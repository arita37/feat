164  SGVector<int32_t> get_support_vectors() const
165  {
166  if (!m_train_state)
167  SG_ERROR("Please enable save_train_state option and train machine.\n")
168 
169  ASSERT(m_labels && m_labels->get_label_type() == LT_MULTICLASS)
170 
171  int32_t num_vectors = m_features->get_num_vectors();
172  int32_t num_classes = ((CMulticlassLabels*) m_labels)->get_num_classes();
173 
174  v_array<int32_t> nz_idxs;
175  nz_idxs.reserve(num_vectors);
176 
177  for (int32_t i=0; i<num_vectors; i++)
178  {
179  for (int32_t y=0; y<num_classes; y++)
180  {
181  if (CMath::abs(m_train_state->alpha[i*num_classes+y])>1e-6)
182  {
183  nz_idxs.push(i);
184  break;
185  }
186  }
187  }
188  int32_t num_nz = nz_idxs.index();
189  nz_idxs.reserve(num_nz);
190  return SGVector<int32_t>(nz_idxs.begin,num_nz);
191  }

214  bool train_machine(CFeatures* data)
215  {
216  if (data)
217  set_features((CDotFeatures*)data);
218 
219  ASSERT(m_features)
220  ASSERT(m_labels && m_labels->get_label_type()==LT_MULTICLASS)
221  ASSERT(m_multiclass_strategy)
222  init_strategy();
223 
224  int32_t num_vectors = m_features->get_num_vectors();
225  int32_t num_classes = ((CMulticlassLabels*) m_labels)->get_num_classes();
226  int32_t bias_n = m_use_bias ? 1 : 0;
227 
228  liblinear_problem mc_problem;
229  mc_problem.l = num_vectors;
230  mc_problem.n = m_features->get_dim_feature_space() + bias_n;
231  mc_problem.y = SG_MALLOC(float64_t, mc_problem.l);
232  for (int32_t i=0; i<num_vectors; i++)
233  mc_problem.y[i] = ((CMulticlassLabels*) m_labels)->get_int_label(i);
234 
235  mc_problem.x = m_features;
236  mc_problem.use_bias = m_use_bias;
237 
238  SGMatrix<float64_t> w0 = obtain_regularizer_matrix();
239 
240  if (!m_train_state)
241  m_train_state = new mcsvm_state();
242 
243  float64_t* C = SG_MALLOC(float64_t, num_vectors);
244  for (int32_t i=0; i<num_vectors; i++)
245  C[i] = m_C;
246 
247  Solver_MCSVM_CS solver(&mc_problem,num_classes,C,w0.matrix,m_epsilon,
248  m_max_iter,m_max_train_time,m_train_state);
249  solver.solve();
250 
251  m_machines->reset_array();
252  for (int32_t i=0; i<num_classes; i++)
253  {
254  CLinearMachine* machine = new CLinearMachine();
255  SGVector<float64_t> cw(mc_problem.n-bias_n);
256 
257  for (int32_t j=0; j<mc_problem.n-bias_n; j++)
258  cw[j] = m_train_state->w[j*num_classes+i];
259 
260  machine->set_w(cw);
261 
262  if (m_use_bias)
263  machine->set_bias(m_train_state->w[(mc_problem.n-bias_n)*num_classes+i]);
264 
265  m_machines->push_back(machine);
266  }
267 
268  if (!m_save_train_state)
269  reset_train_state();
270 
271  SG_FREE(C);
272  SG_FREE(mc_problem.y);
273 
274  return true;
275  }
