463  int Individual::check_dominance(const Individual& b) const
464  {
465  /* Check whether this individual dominates b.
466  *
467  * Input:
468  *
469  * b: another individual
470  *
471  * Output:
472  *
473  * 1: this individual dominates b; -1: b dominates this; 0: neither dominates
474  */
475 
476  int flag1 = 0, // to check if this has a smaller objective
477  flag2 = 0; // to check if b has a smaller objective
478 
479  for (int i=0; i<obj.size(); ++i) {
480  if (obj[i] < b.obj[i])
481  flag1 = 1;
482  else if (obj[i] > b.obj[i])
483  flag2 = 1;
484  }
485 
486  if (flag1==1 && flag2==0)
487  // there is at least one smaller objective for this and none for b
488  return 1;
489  else if (flag1==0 && flag2==1)
490  // there is at least one smaller objective for b and none for this
491  return -1;
492  else
493  // no smaller objective or both have one smaller
494  return 0;
495 
496  }
