168  for (unsigned i=0; i<p.size(); ++i)
169  p[i] = 1-p[i];

171  p = softmax(p);
172  for (unsigned i=0; i<p.size(); ++i)
173  p[i] = u + fb*(u-p[i]);

303  size_t Individual::subtree(size_t i, char otype='0') const
304  {
305 
320  size_t tmp = i;
321  assert(i>=0 && "attempting to grab subtree with index < 0");
322 
323  if (program[i]->total_arity()==0) // return this index if it is a terminal
324  return i;
325 
326  std::map<char, unsigned int> arity = program[i]->arity;
327 
328  if (otype!='0') // if we are recursing (otype!='0'), we need to find
329  // where the nodes to recurse are.
330  {
331  while (i>0 && program[i]->otype != otype) --i;
332  assert(program[i]->otype == otype && "invalid subtree arguments");
333  }
334 
335  for (unsigned int j = 0; j<arity['f']; ++j)
336  i = subtree(--i,'f'); // recurse for floating arguments
337  size_t i2 = i; // index for second recursion
338  for (unsigned int j = 0; j<arity['b']; ++j)
339  i2 = subtree(--i2,'b');
340  size_t i3 = i2; // recurse for boolean arguments
341  for (unsigned int j = 0; j<arity['z']; ++j)
342  i3 = subtree(--i3,'z');
343  return std::min(i,i3);

370  int Individual::check_dominance(const Individual& b) const
371  {
372  /* Check whether this individual dominates b.
373  *
374  * Input:
375  *
376  * b: another individual
377  *
378  * Output:
379  *
380  * 1: this individual dominates b; -1: b dominates this; 0: neither dominates
381  */
382 
383  int flag1 = 0, // to check if this has a smaller objective
384  flag2 = 0; // to check if b has a smaller objective
385 
386  for (int i=0; i<obj.size(); ++i) {
387  if (obj[i] < b.obj[i])
388  flag1 = 1;
389  else if (obj[i] > b.obj[i])
390  flag2 = 1;
391  }
392 
393  if (flag1==1 && flag2==0)
394  // there is at least one smaller objective for this and none for b
395  return 1;
396  else if (flag1==0 && flag2==1)
397  // there is at least one smaller objective for b and none for this
398  return -1;
399  else
400  // no smaller objective or both have one smaller
401  return 0;
402 
403  }
