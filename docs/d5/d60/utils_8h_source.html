99  vector<size_t> argsort(const vector<T> &v, bool ascending=true)
100  {
101  // initialize original index locations
102  vector<size_t> idx(v.size());
103  std::iota(idx.begin(), idx.end(), 0);
104 
105  // sort indexes based on comparing values in v
106  if (ascending)
107  {
108  sort(idx.begin(), idx.end(),
109  [&v](size_t i1, size_t i2) {return v[i1] < v[i2];});
110  }
111  else
112  {
113  sort(idx.begin(), idx.end(),
114  [&v](size_t i1, size_t i2) {return v[i1] > v[i2];});
115  }
116 
117  return idx;
118  }
